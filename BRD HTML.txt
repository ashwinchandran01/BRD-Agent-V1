            // Remove active class from all tabs and content
            document.querySelectorAll('.tab').forEach(tab => tab.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(content => {
                content.classList.remove('active');
            });
            
            // Find and activate the clicked tab
            const clickedTab = document.querySelector(`[onclick="switchTab('${tabName}')"]`);
            if (clickedTab) {
                clickedTab.classList.add('active');
            }

            const targetTab = document.getElementById(`${tabName}-tab`);
            if (targetTab) {
                targetTab.classList.add('active');

        function handleDragOver(event) {
            event.preventDefault();
        }

        function handleDragEnter(event, element) {
            event.preventDefault();
            element.classList.add('dragover');
        }

        function handleDragLeave(event, element) {
            event.preventDefault();
            if (!element.contains(event.relatedTarget)) {
                element.classList.remove('dragover');
            }
        }

        function handleDrop(event, type) {
            event.preventDefault();
            const uploadArea = event.target.closest('.upload-area');
            uploadArea.classList.remove('dragover');
            const files = Array.from(event.dataTransfer.files);
            addFiles(files, type);
        }

            const files = Array.from(event.target.files);
            addFiles(files, type);

        function addFiles(files, type) {
            // Validate file types and sizes
            const allowedTypes = CONFIG.allowedFileTypes[type];
            const validFiles = [];
            
            files.forEach(file => {
                // Check file size
                if (file.size > CONFIG.maxFileSize) {
                    showNotification(`File "${file.name}" is too large. Max size: ${CONFIG.maxFileSize / 1024 / 1024}MB`, 'error');
                    return;
                }
                
                // Check file type
                const fileExtension = '.' + file.name.split('.').pop().toLowerCase();
                if (!allowedTypes.includes(fileExtension)) {
                    showNotification(`File type "${fileExtension}" not allowed for ${type} documents`, 'error');
                    return;
                }
                
                // Check for duplicates
                if (!uploadedFiles[type].some(f => f.name === file.name && f.size === file.size)) {
                    validFiles.push(file);
                    uploadedFiles[type].push(file);
                }
            });
            
            if (validFiles.length > 0) {
                updateFilePreview(type);
                updateGenerateButton();
                showNotification(`${validFiles.length} file(s) uploaded successfully!`, 'success');
            }
        }

        function updateFilePreview(type) {
            const previewElement = document.getElementById(`${type}-preview`);
            const filesList = document.getElementById(type === 'solution' ? 'solution-files' : 'flowchart-files-list');
            
            if (uploadedFiles[type].length > 0) {
                previewElement.style.display = 'block';
                previewElement.classList.add('fade-in-up');
                filesList.innerHTML = '';
                
                uploadedFiles[type].forEach((file, index) => {
                    const li = document.createElement('li');
                    li.className = 'file-item';
                    li.innerHTML = `
                        <div class="file-info">
                            <span class="file-name">${file.name}</span>
                            <span class="file-size">(${formatFileSize(file.size)})</span>
                        </div>
                        <span class="remove-file" onclick="removeFile('${type}', ${index})">Remove</span>
                    `;
                    filesList.appendChild(li);
                });
            } else {
                previewElement.style.display = 'none';
            }
        }

        function removeFile(type, index) {
            uploadedFiles[type].splice(index, 1);
            updateFilePreview(type);
            updateGenerateButton();
            showNotification('File removed successfully', 'info');
        }

        function formatFileSize(bytes) {
            if (bytes === 0) return '0 Bytes';
            const k = 1024;
            const sizes = ['Bytes', 'KB', 'MB', 'GB'];
            const i = Math.floor(Math.log(bytes) / Math.log(k));
            return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
        }

        function updateGenerateButton() {
            const generateBtn = document.getElementById('generate-btn');
            const hasFiles = uploadedFiles.solution.length > 0 || uploadedFiles.flowchart.length > 0;
            generateBtn.disabled = !hasFiles;
        }

        // BRD Generation with n8n webhook integration
        async function generateBRD() {
            if (uploadedFiles.solution.length === 0 && uploadedFiles.flowchart.length === 0) {
                showNotification('Please upload at least one document', 'error');
                return;
            }

            // Check if webhook URL is configured
            if (CONFIG.n8nWebhookUrl === 'YOUR_N8N_WEBHOOK_URL_HERE') {
                showNotification('‚öôÔ∏è Please configure your n8n webhook URL in the code', 'error');
                return;
            }

            switchTab('progress');
            startProgressSimulation();

            try {
                // Prepare FormData for n8n webhook
                const formData = new FormData();
                
                // Add solution documents
                uploadedFiles.solution.forEach((file, index) => {
                    formData.append(`solution_doc_${index}`, file);
                });
                
                // Add flowchart/visual files
                uploadedFiles.flowchart.forEach((file, index) => {
                    formData.append(`visual_asset_${index}`, file);
                });
                
                // Add metadata
                formData.append('metadata', JSON.stringify({
                    timestamp: new Date().toISOString(),
                    total_solution_docs: uploadedFiles.solution.length,
                    total_visual_assets: uploadedFiles.flowchart.length,
                    user_agent: navigator.userAgent,
                    source: 'gocomet-brd-agent'
                }));

                console.log('Sending request to n8n webhook:', CONFIG.n8nWebhookUrl);

                // Send to n8n webhook
                const response = await fetch(CONFIG.n8nWebhookUrl, {
                    method: 'POST',
                    body: formData,
                    headers: {
                        'Accept': 'application/json',
                    }
                });

                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }

                const result = await response.json();
                console.log('n8n webhook response:', result);
                
                // Complete progress with real result
                completeProgress(result);
                
            } catch (error) {
                console.error('Error calling n8n webhook:', error);
                showNotification(`Failed to process documents: ${error.message}`, 'error');
                resetProgress();
            }
        }

        function startProgressSimulation() {
            currentProgress = 0;
            updateProgressDisplay(0, 'üöÄ Processing Documents...', 'Uploading and analyzing your documents');
            document.getElementById('step-upload').classList.add('active');
            
            processingInterval = setInterval(() => {
                if (currentProgress < 90) {
                    currentProgress += Math.random() * 15;
                    if (currentProgress > 90) currentProgress = 90;
                    
                    let title, description, activeStep;
                    if (currentProgress < 20) {
                        title = 'üì§ Uploading Documents...';
                        description = 'Securely transferring files to AI processing server';
                        activeStep = 'step-upload';
                    } else if (currentProgress < 40) {
                        title = 'üîç Analyzing Content...';
                        description = 'Extracting text and analyzing document structure';
                        activeStep = 'step-analyze';
                    } else if (currentProgress < 70) {
                        title = 'ü§ñ AI Processing...';
                        description = 'Advanced AI is analyzing your documents for insights';
                        activeStep = 'step-ai';
                    } else {
                        title = 'üìù Generating BRD...';
                        description = 'Creating comprehensive Business Requirements Document';
                        activeStep = 'step-generate';
                    }
                    
                    updateProgressDisplay(currentProgress, title, description);
                    updateProgressSteps(activeStep);
                }
            }, 800);
        }

        function completeProgress(result) {
            clearInterval(processingInterval);
            currentProgress = 100;
            updateProgressDisplay(100, '‚úÖ BRD Generated Successfully!', 'Your Business Requirements Document is ready for review');
            updateProgressSteps('step-complete');
            
            // Handle the BRD content from n8n response
            let brdContent = '';
            if (result && result.brd_content) {
                brdContent = result.brd_content;
            } else if (result && result.content) {
                brdContent = result.content;
            } else if (result && typeof result === 'string') {
                brdContent = result;
            } else {
                brdContent = `
                    <h3>üéØ Business Requirements Document</h3>
                    <h4>Generated from Your Documents</h4>
                    <p><em>AI processing completed. The full BRD content should appear here based on your n8n workflow response.</em></p>
                    <p><strong>Response received:</strong></p>
                    <pre>${JSON.stringify(result, null, 2)}</pre>
                `;
            }
            
            document.getElementById('brd-content').innerHTML = brdContent;
            
            showNotification('üéâ BRD generated successfully by AI!', 'success');
            
            setTimeout(() => {
                switchTab('canvas');
            }, 2000);
        }

        function resetProgress() {
            clearInterval(processingInterval);
            currentProgress = 0;
            updateProgressDisplay(0, 'Ready to Process', 'Upload your documents to begin AI analysis');
            
            document.querySelectorAll('.step').forEach(step => {
                step.classList.remove('active', 'completed');
            });
        }

        function updateProgressDisplay(progress, title, description) {
            document.querySelector('.progress-circle').style.setProperty('--progress', `${progress}%`);
            document.querySelector('.progress-text').textContent = `${Math.round(progress)}%`;
            document.getElementById('progress-title').textContent = title;
            document.getElementById('progress-description').textContent = description;
        }

        function updateProgressSteps(activeStep) {
            const steps = ['step-upload', 'step-analyze', 'step-ai', 'step-generate', 'step-complete'];
            const activeIndex = steps.indexOf(activeStep);
            
            steps.forEach((step, index) => {
                const element = document.getElementById(step);
                element.classList.remove('active', 'completed');
                
                if (index < activeIndex) {
                    element.classList.add('completed');
                } else if (index === activeIndex) {
                    element.classList.add('active');
                }
            });
        }

                <!DOCTYPE html>
                <html>
                <head>
                    <title>Business Requirements Document</title>
                    <style>
                        body { font-family: Arial, sans-serif; line-height: 1.6; max-width: 800px; margin: 0 auto; padding: 20px; }
                        h3 { color: #112f6b; }
                        h4 { color: #0054ff; margin-top: 20px; }
                    </style>
                </head>
                <body>
                    ${brdContent}
                </body>
                </html>
            `;
            
            const blob = new Blob([htmlContent], { type: 'text/html' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'GoComet_BRD_Document.html';
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            
            showNotification('üíæ BRD downloaded successfully!', 'success');
        }
